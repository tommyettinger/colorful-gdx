011/**
012 * Contains code for manipulating colors as {@code int}, packed {@code float}, and {@link Color} values in the RGB
013 * color space. All of these colors have an alpha channel as well as red, green, and blue; the package name isn't
014 * "rgba" because "ycwcm" and "ipt" don't mention alpha either and they also have that channel. RGB is the standard mode
015 * for colors in libGDX, and conversions to and from RGB are in the ColorTools classes for each other color space (like
016 * {@link com.github.tommyettinger.colorful.ipt.ColorTools#fromRGBA(float, float, float, float)} and
017 * {@link com.github.tommyettinger.colorful.ycwcm.ColorTools#toRGBA(float)}).
018 */
019public class ColorTools {
020        /**
021         * No need to instantiate.
022         */
023        private ColorTools(){
024        }
025        /**
026         * Gets a packed float representation of a color given as 4 float components, here, R (red), G (green), B (blue),
027         * and A (alpha or opacity). As long as you use a batch with {@link Shaders#fragmentShaderRGBA} or
028         * {@link Shaders#fragmentShaderMultiplyRGBA} as its shader, colors passed with
029         * {@link com.badlogic.gdx.graphics.g2d.Batch#setPackedColor(float)} will be interpreted as RGB. All channels range
030         * from 0.0 to 1.0, inclusive. Alpha is the multiplicative opacity of the color.
031         * <br>
032         * This method bit-masks the resulting color's byte values, so any values can technically be given to this as red,
033         * green, and blue, but they will only be reversible from the returned float color to the original R, G, and B
034         * values if the original values were in the range that {@link #red(float)}, {@link #green(float)}, and
035         * {@link #blue(float)} return.
036         *
037         * @param red       0f to 1f, red channel
038         * @param green     0f to 1f, green channel
039         * @param blue      0f to 1f, blue channel
040         * @param alpha     0f to 1f, 0f makes the color transparent and 1f makes it opaque
041         * @return a float encoding a color with the given properties
042         */
043        public static float rgb(float red, float green, float blue, float alpha) {
044                return NumberUtils.intBitsToFloat(((int) (alpha * 255) << 24 & 0xFE000000) | ((int) (blue * 255) << 16 & 0xFF0000)
045                                | ((int) (green * 255) << 8 & 0xFF00) | ((int) (red * 255) & 0xFF));
046        }

049         * Writes an RGB-format packed float color (the format produced by {@link ColorTools#rgb(float, float, float, float)})
050         * into an RGBA8888 Color as used by libGDX (called {@code editing}).
051         * @param editing a libGDX color that will be filled in-place with an RGBA conversion of {@code packed}
052         * @param packed a packed float color, as produced by {@link ColorTools#rgb(float, float, float, float)}
053         * @return an RGBA8888 int color
054         */
055        public static Color toColor(Color editing, final float packed)
056        {
057                Color.abgr8888ToColor(editing, packed);
058                return editing;
059        }
060        /**
061         * Converts a packed float color in the format produced by {@link #rgb(float, float, float, float)} to an
062         * RGBA8888 int. This format of int can be used with Pixmap and in some other places in libGDX.
063         * @param packed a packed float color, as produced by {@link #rgb(float, float, float, float)}
064         * @return an RGBA8888 int color
065         */
066        public static int toRGBA8888(final float packed) {
067                return Integer.reverseBytes(NumberUtils.floatToIntColor(packed));
068        }

076                return NumberUtils.intBitsToFloat(Integer.reverseBytes(rgba) & 0xFEFFFFFF);

088        /**
089         * Takes RGBA components from 0.0 to 1.0 each and converts to a packed float in the RGBA format this uses.
090         * @param r red, from 0.0 to 1.0 (both inclusive)
091         * @param g green, from 0.0 to 1.0 (both inclusive)
092         * @param b blue, from 0.0 to 1.0 (both inclusive)
093         * @param a alpha, from 0.0 to 1.0 (both inclusive)
094         * @return a packed float as RGBA, which this class can use
095         */
096        public static float fromRGBA(final float r, final float g, final float b, final float a) {
097                return NumberUtils.intBitsToFloat(
098                                          Math.min(Math.max((int)(r * 255.999f), 0), 255)
099                                                | Math.min(Math.max((int)(g * 255.999f), 0), 255) << 8
100                                                | Math.min(Math.max((int)(b * 255.999f), 0), 255) << 16
101                                                | ((int)(a * 255f) << 24 & 0xFE000000));
102        }

183        /**
184         * Gets a color as an RGBA packed float given floats representing hue, saturation, lightness, and opacity.
185         * All parameters should normally be between 0 and 1 inclusive, though any hue is tolerated (precision loss may
186         * affect the color if the hue is too large). A hue of 0 is red, progressively higher hue values go to orange,
187         * yellow, green, blue, and purple before wrapping around to red as it approaches 1. A saturation of 0 is grayscale,
188         * a saturation of 1 is brightly colored, and values close to 1 will usually appear more distinct than values close
189         * to 0, especially if the hue is different. A lightness of 0.001f or less is always black (also using a shortcut if
190         * this is the case, respecting opacity), while a lightness of 1f is white. Very bright colors are mostly in a band
191         * of high-saturation where lightness is 0.5f.
192         *
193         * @param hue        0f to 1f, color wheel position
194         * @param saturation 0f to 1f, 0f is grayscale and 1f is brightly colored
195         * @param lightness  0f to 1f, 0f is black and 1f is white
196         * @param opacity    0f to 1f, 0f is fully transparent and 1f is opaque
197         * @return a float encoding a color with the given properties
198         */
199        public static float floatGetHSL(float hue, float saturation, float lightness, float opacity) {
200                if (lightness <= 0.001f) {
201                        return NumberUtils.intBitsToFloat((((int) (opacity * 255f) << 24) & 0xFE000000));
202                } else if (lightness >= 0.999f) {
203                                return NumberUtils.intBitsToFloat((((int) (opacity * 255f) << 24) & 0xFE000000) | 0x00FFFFFF);
204                } else {
205                        return FloatColors.hsl2rgb(hue, saturation, lightness, opacity);
206                }
207        }

209        /**
210         * Gets the saturation of the given encoded color, as a float ranging from 0.0f to 1.0f, inclusive.
211         * @param encoded a color as a packed float that can be obtained by {@link #rgb(float, float, float, float)}
212         * @return the saturation of the color from 0.0 (a grayscale color; inclusive) to 1.0 (a bright color, inclusive)
213         */
214        public static float saturation(final float encoded) {
215                final int decoded = NumberUtils.floatToRawIntBits(encoded);
216                final float r = (decoded & 0xff) / 255f;
217                final float g = (decoded >>> 8 & 0xff) / 255f;
218                final float b = (decoded >>> 16 & 0xff) / 255f;
219                float x, y, w;
220                if(g < b) {
221                        x = b;
222                        y = g;
223                }
224                else {
225                        x = g;
226                        y = b;
227                }
228                if(r < x) {
229                        w = r;
230                }
231                else {
232                        w = x;
233                        x = r;
234                }
235                return x - Math.min(w, y);
236//              float d = x - Math.min(w, y);
237//              float li = x * (1f - 0.5f * d / (x + 1e-10f));
238//              return (x - li); // (Math.min(li, 1f - li) + 1e-10f);
239        }

241        public static float lightness(final float encoded) {
242                final int decoded = NumberUtils.floatToRawIntBits(encoded);
243                final float r = (decoded & 0xff) / 255f;
244                final float g = (decoded >>> 8 & 0xff) / 255f;
245                final float b = (decoded >>> 16 & 0xff) / 255f;
246                float x, y, w;
247                if(g < b) {
248                        x = b;
249                        y = g;
250                }
251                else {
252                        x = g;
253                        y = b;
254                }
255                if(r < x) {
256                        w = r;
257                }
258                else {
259                        w = x;
260                        x = r;
261                }
262                float d = x - Math.min(w, y);
263                return x * (1f - 0.5f * d / (x + 1e-10f));
264        }

266        /**
267         * Gets the hue of the given encoded color, as a float from 0f (inclusive, red and approaching orange if increased)
268         * to 1f (exclusive, red and approaching purple if decreased).
269         * @param encoded a color as a packed float that can be obtained by {@link #rgb(float, float, float, float)}
270         * @return The hue of the color from 0.0 (red, inclusive) towards orange, then yellow, and
271         * eventually to purple before looping back to almost the same red (1.0, exclusive)
272         */
273        public static float hue(final float encoded) {
274                final int decoded = NumberUtils.floatToRawIntBits(encoded);
275                final float r = (decoded & 0xff) / 255f;
276                final float g = (decoded >>> 8 & 0xff) / 255f;
277                final float b = (decoded >>> 16 & 0xff) / 255f;
278                float x, y, z, w;
279                if(g < b) {
280                        x = b;
281                        y = g;
282                        z = -1f;
283                        w = 2f / 3f;
284                }
285                else {
286                        x = g;
287                        y = b;
288                        z = 0f;
289                        w = -1f / 3f;
290                }
291                if(r < x) {
292                        z = w;
293                        w = r;
294                }
295                else {
296                        w = x;
297                        x = r;
298                }
299                float d = x - Math.min(w, y);
300                return Math.abs(z + (w - y) / (6f * d + 1e-10f));
301        }

303        /**
304         * Gets a variation on the packed float color basis as another packed float that has its hue, saturation, lightness,
305         * and opacity adjusted by the specified amounts. Note that this edits the color in HSL space, not RGB! Takes
306         * floats representing the amounts of change to apply to hue, saturation, lightness, and opacity; these can be
307         * between -1f and 1f. Returns a float that can be used as a packed or encoded color with methods like
308         * {@link com.badlogic.gdx.graphics.g2d.Batch#setPackedColor(float)}. The float is likely to be different than the
309         * result of {@link #rgb(float, float, float, float)} unless hue, saturation, lightness, and opacity are all 0. This
310         * won't allocate any objects.
311         * <br>
312         * The parameters this takes all specify additive changes for a color component, clamping the final values so they
313         * can't go above 1 or below 0, with an exception for hue, which can rotate around if lower or higher hues would be
314         * used. As an example, if you give this 0.4f for saturation, and the current color has saturation 0.7f, then the
315         * resulting color will have 1f for saturation. If you gave this -0.1f for saturation and the current color again
316         * has saturation 0.7f, then resulting color will have 0.6f for saturation.
317         *
318         * @param basis      a packed float color that will be used as the starting point to make the next color
319         * @param hue        -1f to 1f, the hue change that can be applied to the new float color (not clamped, wraps)
320         * @param saturation -1f to 1f, the saturation change that can be applied to the new float color
321         * @param light      -1f to 1f, the light/brightness change that can be applied to the new float color
322         * @param opacity    -1f to 1f, the opacity/alpha change that can be applied to the new float color
323         * @return a float encoding a variation of basis with the given changes
324         */
325        public static float toEditedFloat(float basis, float hue, float saturation, float light, float opacity) {
326                final int decoded = NumberUtils.floatToRawIntBits(basis);
327                opacity = Math.min(Math.max(opacity + (decoded >>> 25) * (1f / 127f), 0f), 1f);
328                final float r = (decoded & 0xff) / 255f;
329                final float g = (decoded >>> 8 & 0xff) / 255f;
330                final float b = (decoded >>> 16 & 0xff) / 255f;
331                float x, y, z, w;
332                if(g < b) {
333                        x = b;
334                        y = g;
335                        z = -1f;
336                        w = 2f / 3f;
337                }
338                else {
339                        x = g;
340                        y = b;
341                        z = 0f;
342                        w = -1f / 3f;
343                }
344                if(r < x) {
345                        z = w;
346                        w = r;
347                }
348                else {
349                        w = x;
350                        x = r;
351                }
352                final float d = x - Math.min(w, y);
353                final float lum = x * (1f - 0.5f * d / (x + 1e-10f));
354                hue += Math.abs(z + (w - y) / (6f * d + 1e-10f)) + 1f;
355                saturation += (x - lum) / (Math.min(lum, 1f - lum) + 1e-10f);
356                return FloatColors.hsl2rgb(hue - (int)hue, Math.min(Math.max(saturation, 0f), 1f), Math.min(Math.max(lum + light, 0f), 1f), opacity);
357        }

359        /**
360         * Interpolates from the packed float color start towards white by change. While change should be between 0f (return
361         * start as-is) and 1f (return white), start should be a packed color, as from
362         * {@link #rgb(float, float, float, float)}. This is a good way to reduce allocations of temporary Colors, and
363         * is a little more efficient and clear than using {@link FloatColors#lerpFloatColors(float, float, float)} to lerp towards
364         * white. Unlike {@link FloatColors#lerpFloatColors(float, float, float)}, this keeps the alpha of start as-is.
365         * @see #darken(float, float) the counterpart method that darkens a float color
366         * @param start the starting color as a packed float
367         * @param change how much to go from start toward white, as a float between 0 and 1; higher means closer to white
368         * @return a packed float that represents a color between start and white
369         */
370        public static float lighten(final float start, final float change) {
371                final int s = NumberUtils.floatToRawIntBits(start), r = s & 0xFF, g = s >>> 8 & 0xFF, b = s >>> 16 & 0xFF,
372                                a = s & 0xFE000000;
373                return NumberUtils.intBitsToFloat(
374                                ((int) (r + (0xFF - r) * change) & 0xFF) |
375                                ((int) (g + (0xFF - g) * change) & 0xFF) << 8 |
376                                ((int) (b + (0xFF - b) * change) & 0xFF) << 16 |
377                                a);
378        }

401        /**
402         * Brings the chromatic components of {@code start} closer to grayscale by {@code change} (desaturating them). While
403         * change should be between 0f (return start as-is) and 1f (return fully gray), start should be a packed color, as
404         * from {@link #rgb(float, float, float, float)}. This leaves alpha alone, unlike
405         * {@link #lessenChange(float, float)}, which usually changes Y.
406         * <br>
407         * <a href="http://www.graficaobscura.com/matrix/index.html">The algorithm used is from here</a>.
408         * @see #enrich(float, float) the counterpart method that makes a float color more saturated
409         * @param start the starting color as a packed float
410         * @param change how much to change start to a desaturated color, as a float between 0 and 1; higher means a less saturated result
411         * @return a packed float that represents a color between start and a desaturated color
412         */
413        public static float dullen(final float start, final float change) {
414                final float rc = 0.32627f, gc = 0.3678f, bc = 0.30593001f;
415                final int s = NumberUtils.floatToRawIntBits(start), r = s & 0xFF, g = s >>> 8 & 0xFF, b = s >>> 16 & 0xFF,
416                                a = s & 0xFE000000;
417                final float ch = 1f - change, rw = change * rc, gw = change * gc, bw = change * bc;
418                return NumberUtils.intBitsToFloat(
419                                (int) Math.min(Math.max(r * (rw+ch) + g * rw + b * rw, 0), 255) |
420                                (int) Math.min(Math.max(r * gw + g * (gw+ch) + b * gw, 0), 255) << 8 |
421                                (int) Math.min(Math.max(r * bw + g * bw + b * (bw+ch), 0), 255) << 16 |
422                                a);
423        }

425        /**
426         * Pushes the chromatic components of {@code start} away from grayscale by change (saturating them). While change
427         * should be between 0f (return start as-is) and 1f (return maximally saturated), start should be a packed color, as
428         * from {@link #rgb(float, float, float, float)}.
429         * <br>
430         * <a href="http://www.graficaobscura.com/matrix/index.html">The algorithm used is from here</a>.
431         * @see #dullen(float, float) the counterpart method that makes a float color less saturated
432         * @param start the starting color as a packed float
433         * @param change how much to change start to a saturated color, as a float between 0 and 1; higher means a more saturated result
434         * @return a packed float that represents a color between start and a saturated color
435         */
436        public static float enrich(final float start, final float change) {
437                final float rc = 0.32627f, gc = 0.3678f, bc = 0.30593001f;
438                final int s = NumberUtils.floatToRawIntBits(start), r = s & 0xFF, g = s >>> 8 & 0xFF, b = s >>> 16 & 0xFF,
439                                a = s & 0xFE000000;
440                final float ch = 1f + change, rw = (-change) * rc, gw = (-change) * gc, bw = (-change) * bc;
441                return NumberUtils.intBitsToFloat(
442                                (int) Math.min(Math.max(r * (rw+ch) + g * rw + b * rw, 0), 255) |
443                                (int) Math.min(Math.max(r * gw + g * (gw+ch) + b * gw, 0), 255) << 8 |
444                                (int) Math.min(Math.max(r * bw + g * bw + b * (bw+ch), 0), 255) << 16 |
445                                a);
446        }

448        /**
449         * Interpolates from the packed float color start to increase its red channel by change.
450         * While change must be between 0f (return start as-is) and 1f (return maximum red), start should be a packed
451         * color, as from {@link #rgb(float, float, float, float)}. This is a good way to reduce allocations of temporary
452         * Colors, and is a little more efficient and clear than using
453         * {@link FloatColors#lerpFloatColors(float, float, float)} to lerp towards a more-red color. Unlike
454         * {@link FloatColors#lerpFloatColors(float, float, float)}, this keeps the other channels of start as-is.
455         * @see #lowerB(float, float) the counterpart method that makes a float color less red
456         * @param start the starting color as a packed float
457         * @param change how much to change start to a more-red color, as a float between 0 and 1; higher means more red
458         * @return a packed float that represents a color between start and a more red color
459         */
460        public static float raiseR(final float start, final float change) {
461                final int s = NumberUtils.floatToRawIntBits(start), t = s & 0xFF, other = s & 0xFEFFFF00;
462                return NumberUtils.intBitsToFloat(((int) (t + (0xFF - t) * change) & 0xFF) | other);
463        }

465        /**

583         * Given a packed float RGBA color {@code mainColor} and another RGBA color that it should be made to contrast with,
584         * gets a packed float RGBA color with roughly inverted lightness (how the Oklab color space interprets it), but the
585         * same general hue and saturation unless the lightness gets too close to white or black. This won't ever produce
586         * black or other very dark colors, and also has a gap in the range it produces for lightness values between 0.5 and
587         * 0.55. That allows most of the colors this method produces to contrast well as a foreground when displayed on a
588         * background of {@code contrastingColor}, or vice versa. This will leave the lightness unchanged if the
589         * hue/saturation of the contrastingColor and those of the mainColor are already very different. This has nothing to
590         * do with the contrast channel of the tweak in ColorfulBatch; where that part of the tweak can make too-similar
591         * lightness values further apart by just a little, this makes a modification on {@code mainColor} to maximize its
592         * lightness difference from {@code contrastingColor} without losing its other qualities.
593         * @param mainColor a packed float color, as produced by {@link #rgb(float, float, float, float)}; this is the color that will be adjusted
594         * @param contrastingColor a packed float color, as produced by {@link #rgb(float, float, float, float)}; the adjusted mainColor will contrast with this
595         * @return a different RGBA packed float color, based on mainColor but with potentially very different lightness
596         */
597        public static float inverseLightness(final float mainColor, final float contrastingColor)
598        {
599                return com.github.tommyettinger.colorful.oklab.ColorTools.toRGBA(
600                                com.github.tommyettinger.colorful.oklab.ColorTools.inverseLightness(
601                                                com.github.tommyettinger.colorful.oklab.ColorTools.fromRGBA(mainColor),
602                                                com.github.tommyettinger.colorful.oklab.ColorTools.fromRGBA(contrastingColor)));
603        }

607         * with, gets a packed float RGBA color with lightness that should be quite different from
608         * {@code contrastingColor}'s lightness, but the same chromatic channels and opacity. This allows most of the colors

611         * <br>
612         * This is similar to {@link #inverseLightness(float, float)}, but is considerably simpler, and this method will
613         * change the lightness of mainColor when the two given colors have close lightness but distant chroma. Because it
614         * averages the original lightness of mainColor with the modified one, this tends to produce softer color changes.
615         * @param mainColor a packed RGBA float color; this is the color that will be adjusted
616         * @param contrastingColor a packed RGBA float color; the adjusted mainColor will contrast with the lightness of this
617         * @return a different packed RGBA float color, based on mainColor but typically with different lightness

619        public static float differentiateLightness(final float mainColor, final float contrastingColor)
620        {
621                final int main = NumberUtils.floatToRawIntBits(com.github.tommyettinger.colorful.oklab.ColorTools.fromRGBA(mainColor)),

639         * Makes the additive RGBA color stored in {@code color} cause less of a change when used as a tint, as if it were
640         * mixed with neutral gray. When {@code fraction} is 1.0, this returns color unchanged; when fraction is 0.0, it
641         * returns {@link Palette#GRAY}, and when it is in-between 0.0 and 1.0 it returns something between the two. This is

643         * @param color a color that should have its tinting effect potentially weakened
644         * @param fraction how much of {@code color} should be kept, from 0.0 to 1.0

647        public static float lessenChange(final float color, float fraction) {

675         * Given a packed float RGB color, this edits its red, green, blue, and alpha channels by first multiplying each
676         * channel by the corresponding "mul" parameter and then adding the corresponding "add" parameter, before clamping.
677         * This means the red value {@code R} (which is considered in the 0 to 1 range) is multiplied by {@code mulR}, then

679         * value (of course, the given float can't be edited in-place). You can give a value of 0 for any "add" parameter
680         * you want to stay unchanged, or a value of 1 for any "mul" parameter that shouldn't change.
681         * @param encoded a packed float RGB color
682         * @param addR how much to add to the R channel; typically in the -1 to 1 range
683         * @param addG how much to add to the G channel; typically in the -1 to 1 range
684         * @param addB how much to add to the B channel; typically in the -1 to 1 range
685         * @param addAlpha how much to add to the alpha channel; typically in the -1 to 1 range
686         * @param mulR how much to multiply the R channel by; should be non-negative
687         * @param mulG how much to multiply the G channel by; should be non-negative
688         * @param mulB how much to multiply the B channel by; should be non-negative
689         * @param mulAlpha how much to multiply the alpha channel by; should be non-negative
690         * @return a packed float RGB color with the requested edits applied to {@code encoded}
691         */

693                                                                float mulR, float mulG, float mulB, float mulAlpha){
694                final int s = NumberUtils.floatToRawIntBits(encoded), r = s & 0xFF, g = s >>> 8 & 0xFF, b = s >>> 16 & 0xFF,
695                                a = s >>> 25;
696                return NumberUtils.intBitsToFloat(
697                                Math.max(0, Math.min(255, (int) (r * mulR + addR * 255.999f))) |

703        /**
704         * Makes a quasi-randomly-edited variant on the given {@code color}, allowing typically a small amount of
705         * {@code variance} (such as 0.05 to 0.25) between the given color and what this can return. The {@code seed} should
706         * be different each time this is called, and can be obtained from a random number generator to make the colors more
707         * random, or can be incremented on each call. If the seed is only incremented or decremented, then this shouldn't
708         * produce two similar colors in a row unless variance is very small. The variance affects the R, G, and B of the
709         * generated color equally, and each of those channels can go up or down by the given variance as long as the total
710         * distance isn't greater than the variance.
711         * @param color a packed float color, as produced by {@link #rgb(float, float, float, float)}
712         * @param seed a long seed that should be different on each call; should not be 0
713         * @param variance max amount of difference between the given color and the generated color; always less than 1
714         * @return a generated packed float color that should be at least somewhat different from {@code color}
715         */
716        public static float randomEdit(final float color, long seed, final float variance) {
717                final int decoded = NumberUtils.floatToRawIntBits(color);
718                final float r = (decoded & 0xff) / 255f;
719                final float g = (decoded >>> 8 & 0xff) / 255f;
720                final float b = (decoded >>> 16 & 0xff) / 255f;
721                final float limit = variance * variance;
722                float dist, x, y, z;
723                for (int j = 0; j < 50; j++) {
724                        x = (((seed * 0xD1B54A32D192ED03L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
725                        y = (((seed * 0xABC98388FB8FAC03L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
726                        z = (((seed * 0x8CB92BA72F3D8DD7L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
727                        seed += 0x9E3779B97F4A7C15L;
728                        dist = x * x + y * y + z * z;
729                        x += r;
730                        y += g;
731                        z += b;
732                        if(dist <= limit)
733                                return NumberUtils.intBitsToFloat((decoded & 0xFE000000) | ((int)(z * 255.999f) << 16 & 0xFF0000)
734                                        | ((int)(y * 255.999f) << 8 & 0xFF00) | (int)(x * 255.999f));
735                }
736                return color;
737        }
